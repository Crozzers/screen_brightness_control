 22def get_brightness(
 23    display: Optional[DisplayIdentifier] = None,
 24    method: Optional[str] = None,
 25    allow_duplicates: Optional[bool] = None,
 26    verbose_error: bool = False
 27) -> List[Union[IntPercentage, None]]:
 28    '''
 29    Returns the current brightness of one or more displays
 30
 31    Args:
 32        display (.types.DisplayIdentifier): the specific display to query
 33        method: the method to use to get the brightness. See `get_methods` for
 34            more info on available methods
 35        allow_duplicates: controls whether to filter out duplicate displays or not.
 36        verbose_error: controls the level of detail in the error messages
 37
 38    Returns:
 39        A list of `.types.IntPercentage` values, each being the brightness of an
 40        individual display. Invalid displays may return None.
 41
 42    Example:
 43        ```python
 44        import screen_brightness_control as sbc
 45
 46        # get the current screen brightness (for all detected displays)
 47        current_brightness = sbc.get_brightness()
 48
 49        # get the brightness of the primary display
 50        primary_brightness = sbc.get_brightness(display=0)
 51
 52        # get the brightness of the secondary display (if connected)
 53        secondary_brightness = sbc.get_brightness(display=1)
 54        ```
 55    '''
 56    result = __brightness(
 57        display=display,
 58        method=method,
 59        meta_method='get',
 60        allow_duplicates=allow_duplicates,
 61        verbose_error=verbose_error
 62    )
 63    # __brightness can return None depending on the `no_return` kwarg. That obviously would never happen here
 64    # but the type checker doesn't see it that way.
 65    return [] if result is None else result

 68@config.default_params
 69def set_brightness(
 70    value: Percentage,
 71    display: Optional[DisplayIdentifier] = None,
 72    method: Optional[str] = None,
 73    force: bool = False,
 74    allow_duplicates: Optional[bool] = None,
 75    verbose_error: bool = False,
 76    no_return: bool = True
 77) -> Optional[List[Union[IntPercentage, None]]]:
 78    '''
 79    Sets the brightness level of one or more displays to a given value.
 80
 81    Args:
 82        value (.types.Percentage): the new brightness level
 83        display (.types.DisplayIdentifier): the specific display to adjust
 84        method: the method to use to set the brightness. See `get_methods` for
 85            more info on available methods
 86        force: [*Linux Only*] if False the brightness will never be set lower than 1.
 87            This is because on most displays a brightness of 0 will turn off the backlight.
 88            If True, this check is bypassed
 89        allow_duplicates: controls whether to filter out duplicate displays or not.
 90        verbose_error: boolean value controls the amount of detail error messages will contain
 91        no_return: don't return the new brightness level(s)
 92
 93    Returns:
 94        If `no_return` is set to `True` (the default) then this function returns nothing.
 95        Otherwise, a list of `.types.IntPercentage` is returned, each item being the new
 96        brightness of each adjusted display (invalid displays may return None)
 97
 98    Example:
 99        ```python
100        import screen_brightness_control as sbc
101
102        # set brightness to 50%
103        sbc.set_brightness(50)
104
105        # set brightness to 0%
106        sbc.set_brightness(0, force=True)
107
108        # increase brightness by 25%
109        sbc.set_brightness('+25')
110
111        # decrease brightness by 30%
112        sbc.set_brightness('-30')
113
114        # set the brightness of display 0 to 50%
115        sbc.set_brightness(50, display=0)
116        ```
117    '''
118    if isinstance(value, str) and ('+' in value or '-' in value):
119        output: List[Union[IntPercentage, None]] = []
120        for monitor in filter_monitors(display=display, method=method, allow_duplicates=allow_duplicates):
121            # `filter_monitors()` will raise an error if no valid displays are found
122            display_instance = Display.from_dict(monitor)
123            display_instance.set_brightness(value=value, force=force)
124            output.append(None if no_return else display_instance.get_brightness())
125
126        return None if no_return else output
127
128    if platform.system() == 'Linux' and not force:
129        lower_bound = 1
130    else:
131        lower_bound = 0
132
133    value = percentage(value, lower_bound=lower_bound)
134
135    return __brightness(
136        value, display=display, method=method,
137        meta_method='set', no_return=no_return,
138        allow_duplicates=allow_duplicates,
139        verbose_error=verbose_error
140    )

143@config.default_params
144def fade_brightness(
145    finish: Percentage,
146    start: Optional[Percentage] = None,
147    interval: float = 0.01,
148    increment: int = 1,
149    blocking: bool = True,
150    force: bool = False,
151    logarithmic: bool = True,
152    stoppable: bool = True,
153    **kwargs
154) -> Union[List[threading.Thread], List[Union[IntPercentage, None]]]:
155    '''
156    Gradually change the brightness of one or more displays
157
158    Args:
159        finish (.types.Percentage): fade to this brightness level
160        start (.types.Percentage): where the brightness should fade from.
161            If this arg is not specified, the fade will be started from the
162            current brightness.
163        interval: the time delay between each step in brightness
164        increment: the amount to change the brightness by per step
165        blocking: whether this should occur in the main thread (`True`) or a new daemonic thread (`False`)
166        force: [*Linux Only*] if False the brightness will never be set lower than 1.
167            This is because on most displays a brightness of 0 will turn off the backlight.
168            If True, this check is bypassed
169        logarithmic: follow a logarithmic brightness curve when adjusting the brightness
170        stoppable: whether the fade can be stopped by starting a new fade on the same display
171        **kwargs: passed through to `filter_monitors` for display selection.
172            Will also be passed to `get_brightness` if `blocking is True`
173
174    Returns:
175        By default, this function calls `get_brightness()` to return the new
176        brightness of any adjusted displays.
177
178        If `blocking` is set to `False`, then a list of threads are
179        returned, one for each display being faded.
180
181    Example:
182        ```python
183        import screen_brightness_control as sbc
184
185        # fade brightness from the current brightness to 50%
186        sbc.fade_brightness(50)
187
188        # fade the brightness from 25% to 75%
189        sbc.fade_brightness(75, start=25)
190
191        # fade the brightness from the current value to 100% in steps of 10%
192        sbc.fade_brightness(100, increment=10)
193
194        # fade the brightness from 100% to 90% with time intervals of 0.1 seconds
195        sbc.fade_brightness(90, start=100, interval=0.1)
196
197        # fade the brightness to 100% in a new thread
198        sbc.fade_brightness(100, blocking=False)
199        ```
200    '''
201    # make sure only compatible kwargs are passed to filter_monitors
202    available_monitors = filter_monitors(
203        **{k: v for k, v in kwargs.items() if k in (
204            'display', 'haystack', 'method', 'include', 'allow_duplicates'
205        )}
206    )
207
208    threads = []
209    for i in available_monitors:
210        display = Display.from_dict(i)
211
212        thread = threading.Thread(target=display._fade_brightness, args=(finish,), kwargs={
213            'start': start,
214            'interval': interval,
215            'increment': increment,
216            'force': force,
217            'logarithmic': logarithmic,
218            'stoppable': stoppable
219        })
220        thread.start()
221        threads.append(thread)
222
223    if not blocking:
224        return threads
225
226    for t in threads:
227        t.join()
228    return get_brightness(**kwargs)

296def get_methods(name: Optional[str] = None) -> Dict[str, Type[BrightnessMethod]]:
297    '''
298    Returns all available brightness method names and their associated classes.
299
300    Args:
301        name: if specified, return the method corresponding to this name
302
303    Raises:
304        ValueError: if the given name is incorrect
305
306    Example:
307        ```python
308        import screen_brightness_control as sbc
309
310        all_methods = sbc.get_methods()
311
312        for method_name, method_class in all_methods.items():
313            print('Method:', method_name)
314            print('Class:', method_class)
315            print('Associated monitors:', sbc.list_monitors(method=method_name))
316        ```
317    '''
318    methods = {i.__name__.lower(): i for i in _OS_METHODS}
319
320    if name is None:
321        return methods
322
323    if not isinstance(name, str):
324        raise TypeError(f'name must be of type str, not {type(name)!r}')
325
326    name = name.lower()
327    if name in methods:
328        return {name: methods[name]}
329
330    _logger.debug(f'requested method {name!r} invalid')
331    raise ValueError(
332        f'invalid method {name!r}, must be one of: {list(methods)}')

369    def fade_brightness(
370        self,
371        finish: Percentage,
372        start: Optional[Percentage] = None,
373        interval: float = 0.01,
374        increment: int = 1,
375        force: bool = False,
376        logarithmic: bool = True,
377        blocking: bool = True,
378        stoppable: bool = True
379    ) -> Optional[threading.Thread]:
380        '''
381        Gradually change the brightness of this display to a set value.
382        Can execute in the current thread, blocking until completion,
383        or in a separate thread, allowing concurrent operations.
384        When set as non-blocking and stoppable, a new fade can halt the this operation.
385
386        Args:
387            finish (.types.Percentage): the brightness level to end up on
388            start (.types.Percentage): where the fade should start from. Defaults
389                to whatever the current brightness level for the display is
390            interval: time delay between each change in brightness
391            increment: amount to change the brightness by each time (as a percentage)
392            force: [*Linux only*] allow the brightness to be set to 0. By default,
393                brightness values will never be set lower than 1, since setting them to 0
394                often turns off the backlight
395            logarithmic: follow a logarithmic curve when setting brightness values.
396                See `logarithmic_range` for rationale
397            blocking: run this function in the current thread and block until it completes
398            stoppable: whether this fade will be stopped by starting a new fade on the same display
399
400        Returns:
401            If `blocking` is `False`, returns a `threading.Thread` object representing the
402            thread in which the fade operation is running.
403            Otherwise, it returns None.
404        '''
405        thread = threading.Thread(target=self._fade_brightness, args=(finish,), kwargs={
406            'start': start,
407            'interval': interval,
408            'increment': increment,
409            'force': force,
410            'logarithmic': logarithmic,
411            'stoppable': stoppable
412        })
413        thread.start()
414
415        if not blocking:
416            return thread
417        else:
418            thread.join()
419            return None

421    def _fade_brightness(
422        self,
423        finish: Percentage,
424        start: Optional[Percentage] = None,
425        interval: float = 0.01,
426        increment: int = 1,
427        force: bool = False,
428        logarithmic: bool = True,
429        stoppable: bool = True
430    ) -> None:
431        # Record the latest thread for this display so that other stoppable threads can be stopped
432        display_key = frozenset((self.method, self.index))
433        self._fade_thread_dict[display_key] = threading.current_thread()
434        # minimum brightness value
435        if platform.system() == 'Linux' and not force:
436            lower_bound = 1
437        else:
438            lower_bound = 0
439
440        current = self.get_brightness()
441
442        finish = percentage(finish, current, lower_bound)
443        start = percentage(
444            current if start is None else start, current, lower_bound)
445
446        # mypy says "object is not callable" but range is. Ignore this
447        range_func: Callable = logarithmic_range if logarithmic else range  # type: ignore[assignment]
448        increment = abs(increment)
449        if start > finish:
450            increment = -increment
451
452        self._logger.debug(
453            f'fade {start}->{finish}:{increment}:logarithmic={logarithmic}')
454
455        # Record the time when the next brightness change should start
456        next_change_start_time = time.time()
457        for value in range_func(start, finish, increment):
458            if stoppable and threading.current_thread() != self._fade_thread_dict[display_key]:
459                # If the current thread is stoppable and it's not the latest thread, stop fading
460                break
461            # `value` is ensured not to hit `finish` in loop, this will be handled in the final step.
462            self.set_brightness(value, force=force)
463
464            # `interval` is the intended time between the start of each brightness change.
465            next_change_start_time += interval
466            sleep_time = next_change_start_time - time.time()
467            # Skip sleep if the scheduled time has already passed
468            if sleep_time > 0:
469                time.sleep(sleep_time)
470        else:
471            # As `value` doesn't hit `finish` in loop, we explicitly set brightness to `finish`.
472            # This also avoids an unnecessary sleep in the last iteration.
473            if not stoppable or threading.current_thread() == self._fade_thread_dict[display_key]:
474                self.set_brightness(finish, force=force)

558@config.default_params
559def filter_monitors(
560    display: Optional[DisplayIdentifier] = None,
561    haystack: Optional[List[dict]] = None,
562    method: Optional[str] = None,
563    include: List[str] = [],
564    allow_duplicates: Optional[bool] = None
565) -> List[dict]:
566    '''
567    Searches through the information for all detected displays
568    and attempts to return the info matching the value given.
569    Will attempt to match against index, name, edid, method and serial
570
571    Args:
572        display (.types.DisplayIdentifier): the display you are searching for
573        haystack: the information to filter from.
574            If this isn't set it defaults to the return of `list_monitors_info`
575        method: the method the monitors use. See `get_methods` for
576            more info on available methods
577        include: extra fields of information to sort by
578        allow_duplicates: controls whether to filter out duplicate displays or not
579
580    Raises:
581        NoValidDisplayError: if the display does not have a match
582        TypeError: if the `display` kwarg is not `int` or `str`
583
584    Example:
585        ```python
586        import screen_brightness_control as sbc
587
588        search = 'GL2450H'
589        match = sbc.filter_displays(search)
590        print(match)
591        # EG output: [{'name': 'BenQ GL2450H', 'model': 'GL2450H', ... }]
592        ```
593    '''
594    if display is not None and type(display) not in (str, int):
595        raise TypeError(
596            f'display kwarg must be int or str, not "{type(display).__name__}"')
597
598    def get_monitor_list():
599        # if we have been provided with a list of monitors to sift through then use that
600        # otherwise, get the info ourselves
601        if haystack is not None:
602            monitors_with_duplicates = haystack
603            if method is not None:
604                method_class = next(iter(get_methods(method).values()))
605                monitors_with_duplicates = [
606                    i for i in haystack if i['method'] == method_class]
607        else:
608            monitors_with_duplicates = list_monitors_info(
609                method=method, allow_duplicates=True)
610
611        return monitors_with_duplicates
612
613    def filter_monitor_list(to_filter):
614        # This loop does two things:
615        # 1. Filters out duplicate monitors
616        # 2. Matches the display kwarg (if applicable)
617
618        # When duplicates are allowed, the logic is straightforward:
619        if allow_duplicates:
620            if display is None:
621                # no monitor should be filtered out
622                return to_filter
623            elif isinstance(display, int):
624                # 'display' variable should be the index of the monitor
625                # return a list with the monitor at the index or an empty list if the index is out of range
626                return to_filter[display:display + 1]
627            elif isinstance(display, str):
628                # 'display' variable should be an identifier of the monitor
629                # multiple monitors with the same identifier are allowed here, so return all of them
630                monitors = []
631                for monitor in to_filter:
632                    for identifier in ['edid', 'serial', 'name'] + include:
633                        if display == monitor.get(identifier, None):
634                            monitors.append(monitor)
635                            break
636                return monitors
637
638        filtered_displays = {}
639        for monitor in to_filter:
640            # find a valid identifier for a monitor, excluding any which are equal to None
641            added = False
642            for identifier in ['edid', 'serial', 'name'] + include:
643                if monitor.get(identifier, None) is None:
644                    continue
645
646                m_id = monitor[identifier]
647                if m_id in filtered_displays:
648                    break
649
650                if isinstance(display, str) and m_id != display:
651                    continue
652
653                # check we haven't already added the monitor
654                if not added:
655                    filtered_displays[m_id] = monitor
656                    added = True
657
658                # if the display kwarg is an integer and we are currently at that index
659                if isinstance(display, int) and len(filtered_displays) - 1 == display:
660                    return [monitor]
661
662                if added:
663                    break
664        return list(filtered_displays.values())
665
666    duplicates = []
667    for _ in range(3):
668        duplicates = get_monitor_list()
669        if duplicates:
670            break
671        time.sleep(0.4)
672    else:
673        msg = 'no displays detected'

691def __brightness(
692    *args: Any,
693    display: Optional[DisplayIdentifier] = None,
694    method: Optional[str] = None,
695    meta_method: str = 'get',
696    no_return: bool = False,
697    allow_duplicates: bool = False,
698    verbose_error: bool = False,
699    **kwargs: Any
700) -> Optional[List[Union[IntPercentage, None]]]:
701    '''Internal function used to get/set brightness'''
702    _logger.debug(
703        f"brightness {meta_method} request display {display} with method {method}")
704
705    output: List[Union[int, None]] = []
706    errors = []
707
708    for monitor in filter_monitors(display=display, method=method, allow_duplicates=allow_duplicates):
709        try:
710            if meta_method == 'set':
711                monitor['method'].set_brightness(
712                    *args, display=monitor['index'], **kwargs)
713                if no_return:
714                    output.append(None)
715                    continue
716
717            output += monitor['method'].get_brightness(
718                display=monitor['index'], **kwargs)
719        except Exception as e:
720            output.append(None)
721            errors.append((
722                monitor, e.__class__.__name__,
723                traceback.format_exc() if verbose_error else e
724            ))
725
726    if output:
727        output_is_none = set(output) == {None}
728        if (
729            # can't have None output if we are trying to get the brightness
730            (meta_method == 'get' and not output_is_none)
731            or (
732                # if we are setting the brightness then we CAN have a None output
733                # but only if no_return is True.
734                meta_method == 'set'
735                and ((no_return and output_is_none) or not output_is_none)
736            )
737        ):
738            return None if no_return else output
739
740    # if the function hasn't returned then it has failed
741    msg = '\n'
742    if errors:
743        for monitor, exc_name, exc in errors:
744            if isinstance(monitor, str):
745                msg += f'\t{monitor}'
746            else:
747                msg += f'\t{monitor["name"]} ({monitor["serial"]})'
748            msg += f' -> {exc_name}: '
749            msg += str(exc).replace('\n', '\n\t\t') + '\n'
750    else:
751        msg += '\tno valid output was received from brightness methods'
752
753    raise ScreenBrightnessError(msg)

445            current if start is None else start, current, lower_bound)
446
447        # mypy says "object is not callable" but range is. Ignore this
448        range_func: Callable = logarithmic_range if logarithmic else range  # type: ignore[assignment]
449        increment = abs(increment)
450        if start > finish:
451            increment = -increment
452
453        self._logger.debug(
454            f'fade {start}->{finish}:{increment}:logarithmic={logarithmic}')
455
456        # Record the time when the next brightness change should start
457        next_change_start_time = time.time()
458        for value in range_func(start, finish, increment):
459            if stoppable and threading.current_thread() != self._fade_thread_dict[display_key]:
460                # If the current thread is stoppable and it's not the latest thread, stop fading
461                break
462            # `value` is ensured not to hit `finish` in loop, this will be handled in the final step.
463            self.set_brightness(value, force=force)
464
465            # `interval` is the intended time between the start of each brightness change.
466            next_change_start_time += interval
467            sleep_time = next_change_start_time - time.time()
468            # Skip sleep if the scheduled time has already passed
469            if sleep_time > 0:
470                time.sleep(sleep_time)
471        else:
472            # As `value` doesn't hit `finish` in loop, we explicitly set brightness to `finish`.
473            # This also avoids an unnecessary sleep in the last iteration.
474            if not stoppable or threading.current_thread() == self._fade_thread_dict[display_key]:
475                self.set_brightness(finish, force=force)

559@config.default_params
560def filter_monitors(
561    display: Optional[DisplayIdentifier] = None,
562    haystack: Optional[List[dict]] = None,
563    method: Optional[str] = None,
564    include: List[str] = [],
565    allow_duplicates: Optional[bool] = None
566) -> List[dict]:
567    '''
568    Searches through the information for all detected displays
569    and attempts to return the info matching the value given.
570    Will attempt to match against index, name, edid, method and serial
571
572    Args:
573        display (.types.DisplayIdentifier): the display you are searching for
574        haystack: the information to filter from.
575            If this isn't set it defaults to the return of `list_monitors_info`
576        method: the method the monitors use. See `get_methods` for
577            more info on available methods
578        include: extra fields of information to sort by
579        allow_duplicates: controls whether to filter out duplicate displays or not
580
581    Raises:
582        NoValidDisplayError: if the display does not have a match
583        TypeError: if the `display` kwarg is not `int` or `str`
584
585    Example:
586        ```python
587        import screen_brightness_control as sbc
588
589        search = 'GL2450H'
590        match = sbc.filter_displays(search)
591        print(match)
592        # EG output: [{'name': 'BenQ GL2450H', 'model': 'GL2450H', ... }]
593        ```
594    '''
595    if display is not None and type(display) not in (str, int):
596        raise TypeError(
597            f'display kwarg must be int or str, not "{type(display).__name__}"')
598
599    def get_monitor_list():
600        # if we have been provided with a list of monitors to sift through then use that
601        # otherwise, get the info ourselves
602        if haystack is not None:
603            monitors_with_duplicates = haystack
604            if method is not None:
605                method_class = next(iter(get_methods(method).values()))
606                monitors_with_duplicates = [
607                    i for i in haystack if i['method'] == method_class]
608        else:
609            monitors_with_duplicates = list_monitors_info(
610                method=method, allow_duplicates=True)
611
612        return monitors_with_duplicates
613
614    def filter_monitor_list(to_filter):
615        # This loop does two things:
616        # 1. Filters out duplicate monitors
617        # 2. Matches the display kwarg (if applicable)
618
619        # When duplicates are allowed, the logic is straightforward:
620        if allow_duplicates:
621            if display is None:
622                # no monitor should be filtered out
623                return to_filter
624            elif isinstance(display, int):
625                # 'display' variable should be the index of the monitor
626                # return a list with the monitor at the index or an empty list if the index is out of range
627                return to_filter[display:display + 1]
628            elif isinstance(display, str):
629                # 'display' variable should be an identifier of the monitor
630                # multiple monitors with the same identifier are allowed here, so return all of them
631                monitors = []
632                for monitor in to_filter:
633                    for identifier in ['edid', 'serial', 'name'] + include:
634                        if display == monitor.get(identifier, None):
635                            monitors.append(monitor)
636                            break
637                return monitors
638
639        filtered_displays = {}
640        for monitor in to_filter:
641            # find a valid identifier for a monitor, excluding any which are equal to None
642            added = False
643            for identifier in ['edid', 'serial', 'name'] + include:
644                if monitor.get(identifier, None) is None:
645                    continue
646
647                m_id = monitor[identifier]
648                if m_id in filtered_displays:
649                    break
650
651                if isinstance(display, str) and m_id != display:
652                    continue
653
654                # check we haven't already added the monitor
655                if not added:
656                    filtered_displays[m_id] = monitor
657                    added = True
658
659                # if the display kwarg is an integer and we are currently at that index
660                if isinstance(display, int) and len(filtered_displays) - 1 == display:
661                    return [monitor]
662
663                if added:
664                    break
665        return list(filtered_displays.values())
666
667    duplicates = []
668    for _ in range(3):
669        duplicates = get_monitor_list()
670        if duplicates:
671            break
672        time.sleep(0.4)
673    else:
674        msg = 'no displays detected'
675        if method is not None:
676            msg += f' with method: {method!r}'
677        raise NoValidDisplayError(msg)
678
679    monitors = filter_monitor_list(duplicates)
680    if not monitors:
681        # if no displays matched the query
682        msg = 'no displays found'
683        if display is not None:
684            msg += f' with name/serial/edid/index of {display!r}'
685        if method is not None:
686            msg += f' with method of {method!r}'
687        raise NoValidDisplayError(msg)
688
689    return monitors
