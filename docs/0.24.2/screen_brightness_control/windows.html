 38@contextmanager
 39def _wmi_init():
 40    '''internal function to create and return a wmi instance'''
 41    com_init = False
 42    try:
 43        yield wmi.WMI(namespace='wmi')
 44    except Exception as e:
 45        # WMI init will fail outside the main thread, or if CoInitialize wasn't called first
 46        _logger.debug(f'WMI init failed ({e!r}). Calling CoInitialize and retrying')
 47        com_init = True
 48        if COM_MODEL is None:
 49            pythoncom.CoInitialize()
 50        else:
 51            pythoncom.CoInitializeEx(COM_MODEL)
 52
 53        yield wmi.WMI(namespace='wmi')
 54
 55    # only uninitialise if we initialised. Avoid cleaning up resources being used by another library
 56    if com_init:
 57        pythoncom.CoUninitialize()

 60def enum_display_devices() -> Generator[win32api.PyDISPLAY_DEVICEType, None, None]:
 61    '''
 62    Yields all display devices connected to the computer
 63    '''
 64    for monitor_enum in win32api.EnumDisplayMonitors():
 65        pyhandle = monitor_enum[0]
 66        monitor_info = win32api.GetMonitorInfo(pyhandle.handle)
 67        for adaptor_index in range(5):
 68            try:
 69                # EDD_GET_DEVICE_INTERFACE_NAME flag to populate DeviceID field
 70                device = win32api.EnumDisplayDevices(monitor_info['Device'], adaptor_index, 1)
 71            except pywintypes.error:
 72                _logger.debug(
 73                    f'failed to get display device {monitor_info["Device"]} on adaptor index {adaptor_index}'
 74                )
 75            else:
 76                yield device
 77                break

 80def get_display_info() -> List[dict]:
 81    '''
 82    Gets information about all connected displays using WMI and win32api
 83
 84    Example:
 85        ```python
 86        import screen_brightness_control as s
 87
 88        info = s.windows.get_display_info()
 89        for display in info:
 90            print(display['name'])
 91        ```
 92    '''
 93    info = __cache__.get('windows_monitors_info_raw')
 94    if info is None:
 95        info = []
 96        # collect all monitor UIDs (derived from DeviceID)
 97        monitor_uids = {}
 98        for device in enum_display_devices():
 99            monitor_uids[device.DeviceID.split('#')[2]] = device
100
101        # gather list of laptop displays to check against later
102        with _wmi_init() as wmi:
103            try:
104                laptop_displays = [i.InstanceName for i in wmi.WmiMonitorBrightness()]
105            except Exception as e:
106                # don't do specific exception classes here because WMI does not play ball with it
107                _logger.warning(f'get_display_info: failed to gather list of laptop displays - {format_exc(e)}')
108                laptop_displays = []
109
110            extras, desktop, laptop = [], 0, 0
111            uid_keys = list(monitor_uids.keys())
112            for monitor in wmi.WmiMonitorDescriptorMethods():
113                model, serial, manufacturer, man_id, edid = None, None, None, None, None
114                instance_name = monitor.InstanceName.replace('_0', '', 1).split('\\')[2]
115                try:
116                    pydevice = monitor_uids[instance_name]
117                except KeyError:
118                    # if laptop display WAS connected but was later put to sleep (#33)
119                    if instance_name in laptop_displays:
120                        laptop += 1
121                    else:
122                        desktop += 1
123                    _logger.warning(
124                        f'display {instance_name!r} is detected but not present in monitor_uids. Maybe it is asleep?'
125                    )
126                    continue
127
128                # get the EDID
129                try:
130                    edid = ''.join(f'{char:02x}' for char in monitor.WmiGetMonitorRawEEdidV1Block(0)[0])
131                    # we do the EDID parsing ourselves because calling wmi.WmiMonitorID
132                    # takes too long
133                    parsed = EDID.parse(edid)
134                    man_id, manufacturer, model, name, serial = parsed
135                    if name is None:
136                        raise EDIDParseError('parsed EDID returned invalid display name')
137                except EDIDParseError as e:
138                    edid = None
139                    _logger.warning(f'exception parsing edid str for {monitor.InstanceName} - {format_exc(e)}')
140                except Exception as e:
141                    edid = None
142                    _logger.error(f'failed to get EDID string for {monitor.InstanceName} - {format_exc(e)}')
143                finally:
144                    if edid is None:
145                        devid = pydevice.DeviceID.split('#')
146                        serial = devid[2]
147                        man_id = devid[1][:3]
148                        model = devid[1][3:] or 'Generic Monitor'
149                        del devid
150                        if brand := _monitor_brand_lookup(man_id):
151                            man_id, manufacturer = brand
152
153                if (serial, model) != (None, None):
154                    data: dict = {
155                        'name': f'{manufacturer} {model}',
156                        'model': model,
157                        'serial': serial,
158                        'manufacturer': manufacturer,
159                        'manufacturer_id': man_id,
160                        'edid': edid,
161                        'uid': uid_match.group(1) if (uid_match := re.search(r'UID(\d+)', instance_name)) else None,
162                    }
163                    if monitor.InstanceName in laptop_displays:
164                        data['index'] = laptop
165                        data['method'] = WMI
166                        laptop += 1
167                    else:
168                        data['method'] = VCP
169                        desktop += 1
170
171                    if instance_name in uid_keys:
172                        # insert the data into the uid_keys list because
173                        # uid_keys has the monitors sorted correctly. This
174                        # means we don't have to re-sort the list later
175                        uid_keys[uid_keys.index(instance_name)] = data
176                    else:
177                        extras.append(data)
178
179        info = uid_keys + extras
180        if desktop:
181            # now make sure desktop monitors have the correct index
182            count = 0
183            for item in info:
184                if item['method'] == VCP:
185                    item['index'] = count
186                    count += 1
187
188        # return info only which has correct data
189        info = [i for i in info if isinstance(i, dict)]
190
191        __cache__.store('windows_monitors_info_raw', info)
192
193    return info

230class VCP(BrightnessMethod):
231    '''Collection of screen brightness related methods using the DDC/CI commands'''

233    _logger = _logger.getChild('VCP')

235    class _PHYSICAL_MONITOR(Structure):
236        '''internal class, do not call'''
237
238        _fields_ = [('handle', HANDLE), ('description', WCHAR * 128)]
239
240    @classmethod
241    def iter_physical_monitors(cls, start: int = 0) -> Generator[HANDLE, None, None]:
242        '''
243        A generator to iterate through all physical monitors
244        and then close them again afterwards, yielding their handles.
245        It is not recommended to use this function unless you are familiar with `ctypes` and `windll`
246
247        Args:
248            start: skip the first X handles
249
250        Raises:
251            ctypes.WinError: upon failure to enumerate through the monitors
252        '''
253        # user index keeps track of valid monitors
254        user_index = 0
255        # monitor index keeps track of valid and pseudo monitors
256        monitor_index = 0
257        display_devices = list(enum_display_devices())
258
259        with _wmi_init() as wmi:
260            try:
261                laptop_displays = [i.InstanceName.replace('_0', '').split('\\')[2] for i in wmi.WmiMonitorBrightness()]
262            except Exception as e:
263                cls._logger.warning(f'failed to gather list of laptop displays - {format_exc(e)}')
264                laptop_displays = []
265
266        for monitor in map(lambda m: m[0].handle, win32api.EnumDisplayMonitors()):
267            # Get physical monitor count
268            count = DWORD()
269            if not windll.dxva2.GetNumberOfPhysicalMonitorsFromHMONITOR(monitor, byref(count)):
270                raise WinError(None, 'call to GetNumberOfPhysicalMonitorsFromHMONITOR returned invalid result')
271            if count.value > 0:
272                # Get physical monitor handles
273                physical_array = (cls._PHYSICAL_MONITOR * count.value)()
274                if not windll.dxva2.GetPhysicalMonitorsFromHMONITOR(monitor, count.value, physical_array):
275                    raise WinError(None, 'call to GetPhysicalMonitorsFromHMONITOR returned invalid result')
276                for item in physical_array:
277                    # check that the monitor is not a pseudo monitor by
278                    # checking its StateFlags for the
279                    # win32con DISPLAY_DEVICE_ATTACHED_TO_DESKTOP flag
280                    if display_devices[monitor_index].StateFlags & win32con.DISPLAY_DEVICE_ATTACHED_TO_DESKTOP:
281                        # check if monitor is actually a laptop display
282                        if display_devices[monitor_index].DeviceID.split('#')[2] not in laptop_displays:
283                            if start is None or user_index >= start:
284                                yield item.handle
285                            # increment user index as a valid monitor was found
286                            user_index += 1
287                    # increment monitor index
288                    monitor_index += 1
289                    windll.dxva2.DestroyPhysicalMonitor(item.handle)
290

298    @classmethod
299    def get_brightness(cls, display: Optional[int] = None, max_tries: int = 50) -> List[IntPercentage]:
300        '''
301        Args:
302            display: the index of the specific display to query.
303                If unspecified, all detected displays are queried
304            max_tries: the maximum allowed number of attempts to
305                read the VCP output from the display
306
307        Returns:
308            See `BrightnessMethod.get_brightness`
309        '''
310        code = BYTE(0x10)
311        values = []
312        start = display if display is not None else 0
313        for index, handle in enumerate(cls.iter_physical_monitors(start=start), start=start):
314            current = __cache__.get(f'vcp_brightness_{index}')
315            if current is None:
316                cur_out = DWORD()
317                attempt = 0  # avoid UnboundLocalError in else clause if max_tries is 0
318                for attempt in range(max_tries):
319                    if windll.dxva2.GetVCPFeatureAndVCPFeatureReply(handle, code, None, byref(cur_out), None):
320                        current = cur_out.value
321                        break
322                    current = None
323                    time.sleep(0.02 if attempt < 20 else 0.1)
324                else:
325                    cls._logger.error(f'failed to get VCP feature reply for display:{index} after {attempt} tries')
326
327            if current is not None:
328                __cache__.store(f'vcp_brightness_{index}', current, expires=0.1)
329                values.append(current)
330
331            if display == index:
332                # if we've got the display we wanted then exit here, no point iterating through all the others.
333                # Cleanup function usually called in iter_physical_monitors won't get called if we break, so call now
334                windll.dxva2.DestroyPhysicalMonitor(handle)
335                break
336
337        return values
338
339    @classmethod
340    def set_brightness(cls, value: IntPercentage, display: Optional[int] = None, max_tries: int = 50):
341        '''
342        Args:
343            value: percentage brightness to set the display to
344            display: The specific display you wish to query.
345            max_tries: the maximum allowed number of attempts to
346                send the VCP input to the display
347        '''
348        __cache__.expire(startswith='vcp_brightness_')
349        code = BYTE(0x10)
350        value_dword = DWORD(value)
351        start = display if display is not None else 0
352        for index, handle in enumerate(cls.iter_physical_monitors(start=start), start=start):
353            attempt = 0  # avoid UnboundLocalError in else clause if max_tries is 0
354            for attempt in range(max_tries):
355                if windll.dxva2.SetVCPFeature(handle, code, value_dword):
356                    break
357                time.sleep(0.02 if attempt < 20 else 0.1)
358            else:
359                cls._logger.error(f'failed to set display:{index}->{value} after {attempt} tries')
360
361            if display == index:
362                # we have the display we wanted, exit and cleanup
363                windll.dxva2.DestroyPhysicalMonitor(handle)
364                break

367def list_monitors_info(
368    method: Optional[str] = None, allow_duplicates: bool = False, unsupported: bool = False
369) -> List[dict]:
370    '''
371    Lists detailed information about all detected displays
372
373    Args:
374        method: the method the display can be addressed by. See `.get_methods`
375            for more info on available methods
376        allow_duplicates: whether to filter out duplicate displays (displays with the same EDID) or not
377        unsupported: include detected displays that are invalid or unsupported.
378            This argument does nothing on Windows
379    '''
380    # no caching here because get_display_info caches its results
381    info = get_display_info()
382
383    all_methods = get_methods(method).values()
384
385    if method is not None:
386        info = [i for i in info if i['method'] in all_methods]
387
388    if allow_duplicates:
389        return info
390
391    try:
392        # use filter_monitors to remove duplicates
393        return filter_monitors(haystack=info)
394    except NoValidDisplayError:
395        return []

  • name (str): the name of the display
  • model (str): the model of the display
  • manufacturer (str): the name of the display manufacturer
  • manufacturer_id (str): the three letter manufacturer code (see MONITOR_MANUFACTURER_CODES)
  • serial (str): the serial of the display OR some other unique identifier
  • edid (str): the EDID string for the display
  • method (BrightnessMethod): the brightness method associated with this display
  • index (int): the index of the display, relative to the brightness method

256        # monitor index keeps track of valid and pseudo monitors
257        monitor_index = 0

262                laptop_displays = [i.InstanceName.replace('_0', '').split('\\')[2] for i in wmi.WmiMonitorBrightness()]

264                cls._logger.warning(f'failed to gather list of laptop displays - {format_exc(e)}')

268            # Get physical monitor count
269            count = DWORD()
270            if not windll.dxva2.GetNumberOfPhysicalMonitorsFromHMONITOR(monitor, byref(count)):
271                raise WinError(None, 'call to GetNumberOfPhysicalMonitorsFromHMONITOR returned invalid result')
272            if count.value > 0:
273                # Get physical monitor handles
274                physical_array = (cls._PHYSICAL_MONITOR * count.value)()
275                if not windll.dxva2.GetPhysicalMonitorsFromHMONITOR(monitor, count.value, physical_array):
276                    raise WinError(None, 'call to GetPhysicalMonitorsFromHMONITOR returned invalid result')
277                for item in physical_array:
278                    # check that the monitor is not a pseudo monitor by
279                    # checking its StateFlags for the

281                    if display_devices[monitor_index].StateFlags & win32con.DISPLAY_DEVICE_ATTACHED_TO_DESKTOP:
282                        # check if monitor is actually a laptop display
283                        if display_devices[monitor_index].DeviceID.split('#')[2] not in laptop_displays:
284                            if start is None or user_index >= start:
285                                yield item.handle
286                            # increment user index as a valid monitor was found
287                            user_index += 1
288                    # increment monitor index
289                    monitor_index += 1

302        Args:

310        '''
311        code = BYTE(0x10)
312        values = []
313        start = display if display is not None else 0
314        for index, handle in enumerate(cls.iter_physical_monitors(start=start), start=start):
315            current = __cache__.get(f'vcp_brightness_{index}')
316            if current is None:
317                cur_out = DWORD()
318                attempt = 0  # avoid UnboundLocalError in else clause if max_tries is 0

324                    time.sleep(0.02 if attempt < 20 else 0.1)
325                else:

327

334                # Cleanup function usually called in iter_physical_monitors won't get called if we break, so call now
335                windll.dxva2.DestroyPhysicalMonitor(handle)
