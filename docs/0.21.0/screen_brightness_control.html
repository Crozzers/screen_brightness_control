 57def set_brightness(
 58    value: Percentage,
 59    display: Optional[DisplayIdentifier] = None,
 60    method: Optional[str] = None,
 61    force: bool = False,
 62    verbose_error: bool = False,
 63    no_return: bool = True
 64) -> Optional[List[IntPercentage]]:
 65    '''
 66    Sets the brightness level of one or more displays to a given value.
 67
 68    Args:
 69        value (types.Percentage): the new brightness level
 70        display (types.DisplayIdentifier): the specific display to adjust
 71        method: the method to use to set the brightness. See `get_methods` for
 72            more info on available methods
 73        force: [*Linux Only*] if False the brightness will never be set lower than 1.
 74            This is because on most displays a brightness of 0 will turn off the backlight.
 75            If True, this check is bypassed
 76        verbose_error: boolean value controls the amount of detail error messages will contain
 77        no_return: don't return the new brightness level(s)
 78
 79    Returns:
 80        If `no_return` is set to `True` (the default) then this function returns nothing.
 81        Otherwise, a list of `types.IntPercentage` is returned, each item being the new
 82        brightness of each adjusted display (invalid displays may return None)
 83
 84    Example:
 85        ```python
 86        import screen_brightness_control as sbc
 87
 88        # set brightness to 50%
 89        sbc.set_brightness(50)
 90
 91        # set brightness to 0%
 92        sbc.set_brightness(0, force=True)
 93
 94        # increase brightness by 25%
 95        sbc.set_brightness('+25')
 96
 97        # decrease brightness by 30%
 98        sbc.set_brightness('-30')
 99
100        # set the brightness of display 0 to 50%
101        sbc.set_brightness(50, display=0)
102        ```
103    '''
104    if isinstance(value, str) and ('+' in value or '-' in value):
105        output = []
106        for monitor in filter_monitors(display=display, method=method):
107            identifier = Monitor.get_identifier(monitor)[1]
108            current_value = get_brightness(display=identifier)[0]
109            result = set_brightness(
110                # don't need to calculate lower bound here because it will be
111                # done by the other path in `set_brightness`
112                percentage(value, current=current_value),
113                display=identifier,
114                force=force,
115                verbose_error=verbose_error,
116                no_return=no_return
117            )
118            if result is None:
119                output.append(result)
120            else:
121                output += result
122
123        return output
124
125    if platform.system() == 'Linux' and not force:
126        lower_bound = 1
127    else:
128        lower_bound = 0
129
130    value = percentage(value, lower_bound=lower_bound)
131
132    return __brightness(
133        value, display=display, method=method,
134        meta_method='set', no_return=no_return,
135        verbose_error=verbose_error

139def fade_brightness(
140    finish: Percentage,
141    start: Optional[Percentage] = None,
142    interval: float = 0.01,
143    increment: int = 1,
144    blocking: bool = True,
145    force: bool = False,
146    logarithmic: bool = True,
147    **kwargs
148) -> Union[List[threading.Thread], List[IntPercentage]]:
149    '''
150    Gradually change the brightness of one or more displays
151
152    Args:
153        finish (types.Percentage): fade to this brightness level
154        start (types.Percentage): where the brightness should fade from.
155            If this arg is not specified, the fade will be started from the
156            current brightness.
157        interval: the time delay between each step in brightness
158        increment: the amount to change the brightness by per step
159        blocking: whether this should occur in the main thread (`True`) or a new daemonic thread (`False`)
160        force: [*Linux Only*] if False the brightness will never be set lower than 1.
161            This is because on most displays a brightness of 0 will turn off the backlight.
162            If True, this check is bypassed
163        logarithmic: follow a logarithmic brightness curve when adjusting the brightness
164        **kwargs: passed through to `filter_monitors` for display selection.
165            Will also be passed to `get_brightness` if `blocking is True`
166
167    Returns:
168        By default, this function calls `get_brightness()` to return the new
169        brightness of any adjusted displays.
170
171        If the `blocking` is set to `False`, then a list of threads are
172        returned, one for each display being faded.
173
174    Example:
175        ```python
176        import screen_brightness_control as sbc
177
178        # fade brightness from the current brightness to 50%
179        sbc.fade_brightness(50)
180
181        # fade the brightness from 25% to 75%
182        sbc.fade_brightness(75, start=25)
183
184        # fade the brightness from the current value to 100% in steps of 10%
185        sbc.fade_brightness(100, increment=10)
186
187        # fade the brightness from 100% to 90% with time intervals of 0.1 seconds
188        sbc.fade_brightness(90, start=100, interval=0.1)
189
190        # fade the brightness to 100% in a new thread
191        sbc.fade_brightness(100, blocking=False)
192        ```
193    '''
194    # make sure only compatible kwargs are passed to filter_monitors
195    available_monitors = filter_monitors(
196        **{k: v for k, v in kwargs.items() if k in (
197            'display', 'haystack', 'method', 'include'
198        )}
199    )
200
201    threads = []
202    for i in available_monitors:
203        monitor = Monitor(i)
204
205        t1 = monitor.fade_brightness(finish, start=start, interval=interval, increment=increment,
206                                     force=force, logarithmic=logarithmic, blocking=False)
207        threads.append(t1)
208
209    if not blocking:
210        return threads
211
212    for t in threads:
213        t.join()
214    return get_brightness(**kwargs)

350    def fade_brightness(
351        self,
352        finish: Percentage,
353        start: Optional[Percentage] = None,
354        interval: float = 0.01,
355        increment: int = 1,
356        force: bool = False,
357        logarithmic: bool = True
358    ) -> IntPercentage:
359        '''
360        Gradually change the brightness of this display to a set value.
361        This works by incrementally changing the brightness until the desired
362        value is reached.
363
364        Args:
365            finish (types.Percentage): the brightness level to end up on
366            start (types.Percentage): where the fade should start from. Defaults
367                to whatever the current brightness level for the display is
368            interval: time delay between each change in brightness
369            increment: amount to change the brightness by each time (as a percentage)
370            force: [*Linux only*] allow the brightness to be set to 0. By default,
371                brightness values will never be set lower than 1, since setting them to 0
372                often turns off the backlight
373            logarithmic: follow a logarithmic curve when setting brightness values.
374                See `logarithmic_range` for rationale
375
376        Returns:
377            The brightness of the display after the fade is complete.
378            See `types.IntPercentage`
379        '''
380        # minimum brightness value
381        if platform.system() == 'Linux' and not force:
382            lower_bound = 1
383        else:
384            lower_bound = 0
385
386        current = self.get_brightness()
387
388        finish = percentage(finish, current, lower_bound)
389        start = percentage(
390            current if start is None else start, current, lower_bound)
391
392        range_func = logarithmic_range if logarithmic else range
393        increment = abs(increment)
394        if start > finish:
395            increment = -increment
396
397        self._logger.debug(
398            f'fade {start}->{finish}:{increment}:logarithmic={logarithmic}')
399
400        for value in range_func(start, finish, increment):
401            self.set_brightness(value)
402            time.sleep(interval)
403
404        if self.get_brightness() != finish:
405            self.set_brightness(finish, no_return=True)
406
407        return self.get_brightness()

750        # 1. Filters out duplicate monitors
751        # 2. Matches the display kwarg (if applicable)
752        filtered_displays = {}
753        for monitor in to_filter:
754            # find a valid identifier for a monitor, excluding any which are equal to None

799            msg += f' with method of {method!r}'
